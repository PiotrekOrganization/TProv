def close
        @item = current_user.items.where(
                :id => params[:item_id],
                :status => 0
        ).first()
        # check if there is still time for this
        if @item.expires < DateTime.current
                @item.status = 8
                @item.save
                @item.order.quantity = @item.order.quantity + 1
                @item.order.save
        else
                @item.status = 5
        end
        @item.save
        redirect_to writer_item_path(@item)
end

def refresh
        @item = current_user.items.where(
                :id => params[:item_id],
                :status => 8
        ).first
        if @item.order.quantity > 0
                @item.order.quantity = @item.order.quantity - 1
                unless @item.order.valid?
                        redirect_to order_path(@item.order)
                        return
                end
                @item.order.save
                @item.status = 0
                @item.price = @item.order.price
                @item.expires = DateTime.current + @item.order.ticket_time.to_i.seconds
                @item.save
        end
        redirect_to writer_item_path(@item)
end
